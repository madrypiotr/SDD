            unhighlightFunction(element);

Template.profileEdit.helpers({
    email: function () {

    isSelected: function (gender) {
        var gen = this.profile.gender;
        if (gen == gender)
            return "checked";
        else
            return "";
    },
    userZwyczajny: function () {
        return this.profile.userType== USERTYPE.CZLONEK ? true : false;

});
