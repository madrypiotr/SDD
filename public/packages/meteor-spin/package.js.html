  git: 'https://github.com/sachag/meteor-spin'

  "spin.js": "2.0.1"
});

Package.onUse(function (api, where) {
  api.versionsFrom(['METEOR@0.9.0', 'METEOR@1.0']);
  api.use([
    'templating',
    'underscore'
  ], 'client');

  api.addFiles([
    '.npm/package/node_modules/spin.js/spin.js',
    'lib/spinner.html',
    'lib/spinner.css',
    'lib/spinner.js'
  ], 'client');
});
